import requests # Import the requests library to handle HTTP requests
import json 

def emotion_detector(text_to_analyze): # takes a string input (text_to_analyze)
 
    # URL of the emotion analysis service 
    url = 'https://sn-watson-emotion.labs.skills.network/v1/watson.runtime.nlp.v1/NlpService/EmotionPredict' 

    # a dictionary with the text for analysis
    myobj = { "raw_document": { "text": text_to_analyze } } 

    # headers required for the API request   
    header = {"grpc-metadata-mm-model-id": "emotion_aggregated-workflow_lang_en_stock"} 

    # Send a POST request to the API with the text and headers
    response = requests.post(url, json = myobj, headers=header) 

    # If the response status code is 400 - all NONE
    if response.status_code == 400:
        result = {
            'anger': None,
            'disgust': None,
            'fear': None,
            'joy': None,
            'sadness': None,
            'dominant_emotion': None
        }		
    else:
        # Parsing the JSON response from the API 
        formatted_response = json.loads(response.text)

        # Extract the emotion scores dictionary
        emotions = formatted_response['emotionPredictions'][0]['emotion']

        # Find the emotion with the highest score
        dominant_emotion = max(emotions, key=emotions.get)
        dominant_score = emotions[dominant_emotion]
 
        # Build the result dictionary with all emotions and dominant emotion label
        result = {
            'anger': emotions.get('anger', 0),
            'disgust': emotions.get('disgust', 0),
            'fear': emotions.get('fear', 0),
            'joy': emotions.get('joy', 0),
            'sadness': emotions.get('sadness', 0),
            'dominant_emotion': dominant_emotion
        }
    return result
